VPM XML 전체 계층 구조 (모든 태그 및 상세 속성 포함)

  • <vpmapp> VPM 설정 루트 노드
    • <vpmxml-info> version="631"
    • <enforcement-point> enabled="false"
    • ▼ 조건 및 대상 정의부 (conditionObjects)
    • <conditionObjects>
      • <ipobject> name="IP객체명" type="1(Src)/2(Dst)" value="IP주소/서브넷" single="true/false"
      • <host-port> name="호스트포트명" h="호스트" p="포트번호"
      • <af-host> name="AF호스트명"
        • <h> 상세 호스트 주소
      • <time> name="시간명" UTC="false" time="0000..2359"
      • <comb-obj> name="복합객체명" t="1(AND)/2(OR)" n-1="부정여부" n-2="부정여부"
        • <c-l-1> n="조건1이름" 첫 번째 묶음
        • <c-l-2> n="조건2이름" 두 번째 묶음
      • <vpm-cat>
        • <node> n="카테고리명" u-l="도메인목록"
      • <categorylist4> name="리스트명"
        • <sel> 선택된 카테고리
          • <i> 개별 항목
          • <ai> a="ID" n="내장이름"
        • <excl> 제외된 카테고리
        • <sel-p> 부모 카테고리 선택
      • <a-url> name="URL명" r="정규표현식" typ="r"
      • <app-url> name="App이름"
        • <selected-applications> 선택된 앱 목록
      • <op-url> name="Operation명"
        • <selected-operations>
          • <item> 세부 동작(예: facebook_login)
      • <protocol> name="프로토콜명" proto="HTTP/HTTPS/FTP 등"
      • <prot-meth> name="메서드명" method="GET/POST 등"
      • <svc-name> name="서비스명" service="정의된서비스"
      • <user-agent-list> name="UA리스트명"
        • <user-agent-item> regex="패턴"
      • <hdr-obj> name="헤더객체명" header="헤더명" regex="정규식"
      • <hdr-act> name="헤더동작명" action="add/remove"
      • <hdr-supp-list> name="지원헤더목록"
        • <i> 개별 헤더 항목
      • <http-req-body> name="요청본문명" max-size="용량(Byte)"
      • <file-download> name="파일객체명"
        • <file-type-info> file-extension="ZIP" use-file-extension="true"
      • <appar-data-type> name="데이터타입명"
        • <adt-info> type="매직넘버/MIME"
      • <scan> name="스캔동작명" type="ICAP_Req/ICAP_Resp"
      • <ssl-fwd-prxy> name="SSL프록시명" action="Intercept/Bypass"
      • <ssl-fwd-prxy522> name="SSL프록시(버전업)"
      • <ssl-fwd-prxy-on-excep> name="예외시SSL처리명"
      • <svr-cert-valdn> name="인증서검증명" action="Verify/Ignore"
      • <disab_ssl_detect> name="SSL탐지우회명"
      • <encrypted-tap> name="암호화탭명"
      • <client-encrypted-tap> name="클라이언트암호화탭명"
      • <group> name="그룹명" type="LDAP/Local"
      • <adm-auth-obj> name="관리자인증객체명"
      • <policy-id> name="정책ID명" cmt="코멘트"
      • <im-alert> name="IM알림명"
      • <im-msg-refl> name="IM메시지반사명"
      • <bw-mgmt> name="대역폭관리명" class="클래스명"
      • <dny-obj> name="거부객체명"
      • <dny-exc> name="거부예외명" text="예외메시지"
      • <htm-ntfy> name="HTML알림명" file="파일명"
      • <trace-obj> name="추적객체명"
      • <acc-log-fac> name="접근로그시설명"
      • <Process-Request> name="요청프로세스명"
      • <guard> name="가드명" 특정 보호 기능
    • ▼ 글로벌 설정부 (configObjects)
    • <configObjects>
      • <dns-rdns> name="DNS설정명"
    • ▼ 정책 실행부 (layers)
    • <layers>
      • <layer> layertype="WebAccessPolicyTable 등" disabled="true/false"
        • <name> 레이어 이름
        • <numRows> 전체 규칙 수
        • <rowItem> num="순서" enabled="true/false"
          • <colItem> 기본 속성(아래 상세 참조)
              ↳ col="열번호(0~7)"
              ↳ id="구분(no, so, de, se, ac, ti, co, tr, ep)"
              ↳ name="조건객체명/문자열"
              ↳ negate="true/false(부정여부)"
              ↳ type="Condition/String"
              ↳ value="값(no일경우 순번)"
          • <colItemV2> 확장된 속성 (colItem과 동일한 구조 + 추가 메타데이터)
      • <raw-cpl> 사용자가 직접 작성한 CPL 코드 블록
